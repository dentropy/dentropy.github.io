## SQL Query

``` sql

select * from ( 
	select 
		count(*) as msg_count,
		channel_id,
		guild_id
	from
		messages_t
	group by channel_id, guild_id
) as channels_grouped_t
join
(
	select
		id,
		channel_name
	from 
		channels_t
) as channel_lookup_t
on channel_lookup_t.id = channels_grouped_t.channel_id
join
(
	select
		id,
		guild_name
	from 
		guilds_t gt 
) as guild_lookup_t
on guild_lookup_t.id = channels_grouped_t.guild_id
order by msg_count desc;

```
